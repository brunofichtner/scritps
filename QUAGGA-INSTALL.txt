Instalação EVE-NG

1 - Abrir pagina eve-ng e realizar o download da versão free.
https://www.eve-ng.net/index.php/download/

2 - Extrair arquivos EVE-COMM-VM-112.zip:
 EVE-COMM-VM-0.vmdk: VHD padrão VMWare;
 EVE-COMM-VM.ovf: configuração recomendada para a VM, abra-o com seu hypervisor (opção importar);
 
3 - Habilitar o suporte a Intel-VT/AMD-V no menu SISMTEAS , para que as máquinas virtuais dentro da VM utilizem CPUs virtuais direto no processador;

4 - iniciar a vm e subir o serviço
verificar interfaces
brctl show
ip a s dev pnet0
ip ro get 1.2.3.4

# Cloud devices
iface eth1 inet manual
auto pnet1
iface pnet1 inet static
        address 10.99.99.1
        netmask 255.255.255.0
    bridge_ports eth1
    bridge_stp off

 ifup --force pnet1


5 - habilitar firewall e ajustes no kernel.
nano /etc/rc.local

#!/bin/sh
# NAT RFC 1918
iptables -t nat -A POSTROUTING -s 10.0.0.0/8 -o pnet0 -j MASQUERADE
iptables -t nat -A POSTROUTING -s 172.16.0.0/12 -o pnet0 -j MASQUERADE
iptables -t nat -A POSTROUTING -s 192.168.0.0/16 -o pnet0 -j MASQUERADE
# NAT RFC 6598
iptables -t nat -A POSTROUTING -s 100.64.0.0/10 -o pnet0 -j MASQUERADE

( echo net.core.rmem_default=31457280
 echo net.core.wmem_default=31457280
 echo net.core.rmem_max=262144
 echo net.core.wmem_max=262144
 echo net.core.netdev_max_backlog=250000
 echo net.core.optmem_max=20480
 echo net.core.default_qdisc=fq
 echo net.core.somaxconn=4096 ) > /etc/sysctl.d/51-net-core.conf
echo net.ipv4.conf.default.forwarding=1 > /etc/sysctl.d/61-default-foward-ipv4.conf
echo net.ipv6.conf.default.forwarding=1 > /etc/sysctl.d/62-default-foward-ipv6.conf
echo net.ipv4.conf.all.forwarding=1 > /etc/sysctl.d/63-all-foward-ipv4.conf
echo net.ipv6.conf.all.forwarding=1 > /etc/sysctl.d/64-all-foward-ipv6.conf
echo net.ipv4.ip_forward=1 > /etc/sysctl.d/71-ipv4-forward.conf
echo net.ipv6.ip_forward=1 > /etc/sysctl.d/72-ipv6-forward.conf
echo vm.swappiness=1 > /etc/sysctl.d/73-swappiness.conf
echo kernel.panic=3 > /etc/sysctl.d/81-panic.conf
echo kernel.threads-max=1031306 > /etc/sysctl.d/82-threads.conf
echo kernel.pid_max=262144 > /etc/sysctl.d/83-pid.conf
echo kernel.msgmax=327680 > /etc/sysctl.d/84-msgmax.conf
echo kernel.msgmnb=655360 > /etc/sysctl.d/85-msgmnb.conf
echo kernel.msgmni=1310720 > /etc/sysctl.d/86-msgmni.conf
# Aplicar
sysctl --system

#executar rc.local
chmod +x /etc/rc.local
cd /etc/
./rc.local

#---------------------------------------------------------- QUAGGA / ZEBRA / OSPF / BGP

# Instalando quagga no Ubuntu

apt-get -y install quagga

cd /etc/quagga/

touch /etc/quagga/babeld.conf
touch /etc/quagga/bgpd.conf
touch /etc/quagga/isisd.conf
touch /etc/quagga/ospf6d.conf
touch /etc/quagga/ospfd.conf
touch /etc/quagga/ripd.conf
touch /etc/quagga/ripngd.conf
touch /etc/quagga/vtysh.conf
touch /etc/quagga/zebra.conf

cat > daemons << EOF
zebra=yes
bgpd=yes
ospfd=yes
ospf6d=yes
ripd=no
ripngd=no
isisd=no
babeld=no
EOF

cat > debian.conf << EOF
vtysh_enable=yes
zebra_options="  --daemon -A 127.0.0.1"
bgpd_options="   --daemon -A 127.0.0.1"
ospfd_options="  --daemon -A 127.0.0.1"
ospf6d_options=" --daemon -A ::1"
ripd_options="   --daemon -A 127.0.0.1"
ripngd_options=" --daemon -A ::1"
isisd_options="  --daemon -A 127.0.0.1"
babeld_options=" --daemon -A 127.0.0.1"
watchquagga_enable=yes
watchquagga_options=(--daemon)
EOF

chown quagga:quagga /etc/quagga/*

x=$(grep VTYSH /etc/environment)
[ "x$x" = "x" ] && echo "VTYSH_PAGER=more" >> /etc/environment
export VTYSH_PAGER=more


cd /etc/quagga

cat > zebra.conf << EOF
interface lo
ip forwarding
EOF

cat > ospfd.conf << EOF
router ospf
 redistribute connected
 redistribute static
EOF

cat > ospf6d.conf << EOF
router ospf6
 redistribute connected
 redistribute static
EOF

service quagga stop
service quagga stop
service quagga start
